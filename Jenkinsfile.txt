pipeline {
    agent any

    stages {
        stage('Build') {
            steps {
                sh 'docker build -t hospital-management-system .'
            }
        }
        stage('Test') {
            steps {
                sh 'docker run hospital-management-system test'
            }
        }
        stage('Deploy') {
            steps {
                script {
                    // Check if the container exists and remove it
                    sh '''
                        if [ $(docker ps -q -f name=hospital-container) ]; then
                            docker stop hospital-container
                            docker rm hospital-container
                        fi
                    '''
                }
                // Run the new container
                sh 'docker run -d --name hospital-container -p 80:80 hospital-management-system'
            }
        }
    }
}
